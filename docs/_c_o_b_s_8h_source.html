83 static size_t decode(const uint8_t* encodedBuffer,
84 size_t size,
85 uint8_t* decodedBuffer)
86 {
87 if (size == 0)
88 return 0;
89
90 size_t read_index = 0;
91 size_t write_index = 0;
92 uint8_t code = 0;
93 uint8_t i = 0;
94
95 while (read_index < size)
96 {
97 code = encodedBuffer[read_index];
98
99 if (read_index + code > size && code != 1)
100 {
101 return 0;
102 }
103
104 read_index++;
105
106 for (i = 1; i < code; i++)
107 {
108 decodedBuffer[write_index++] = encodedBuffer[read_index++];
109 }
110
111 if (code != 0xFF && read_index != size)
112 {
113 decodedBuffer[write_index++] = '\0';
114 }
115 }
116
