173 void update()
174 {
175 if (_stream == nullptr) return;
176
177 while (_stream->available() > 0)
178 {
179 uint8_t data = _stream->read();
180
181 if (data == PacketMarker)
182 {
183 if (_onPacketFunction || _onPacketFunctionWithSender)
184 {
185 uint8_t _decodeBuffer[_receiveBufferIndex];
186
187 size_t numDecoded = EncoderType::decode(_receiveBuffer,
188 _receiveBufferIndex,
189 _decodeBuffer);
190
191 if (_onPacketFunction)
192 {
193 _onPacketFunction(_decodeBuffer, numDecoded);
194 }
195 else if (_onPacketFunctionWithSender)
196 {
197 _onPacketFunctionWithSender(this, _decodeBuffer, numDecoded);
198 }
199 }
200
201 _receiveBufferIndex = 0;
202 }
203 else
204 {
205 if ((_receiveBufferIndex + 1) < BufferSize)
206 {
207 _receiveBuffer[_receiveBufferIndex++] = data;
208 }
209 else
210 {
211 // Error, buffer overflow if we write.
212 }
213 }
214 }
215 }
216
231 void send(const uint8_t* buffer, size_t size) const
232 {
233 if(_stream == nullptr || buffer == nullptr || size == 0) return;
234
235 uint8_t _encodeBuffer[EncoderType::getEncodedBufferSize(size)];
236
237 size_t numEncoded = EncoderType::encode(buffer,
238 size,
239 _encodeBuffer);
240
241 _stream->write(_encodeBuffer, numEncoded);
242 _stream->write(PacketMarker);
243 }
